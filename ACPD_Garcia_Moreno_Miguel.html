import numpy as np
import csv

def csv_to_numpy(csv_file, delimiter=","):
    row_matrix = []

    with open(csv_file, "r" ) as file:
        reader = csv.reader(file, delimiter=delimiter)
        for row_num, row in enumerate(reader):
            if row_num == 0:
                header = row
                continue

            row_list = []

            for col_num, col in enumerate(row):
                row_list.append(int(row[col_num]))
            
            row_matrix.append(row_list)
    np_array = np.array(row_matrix)
    return np_array

import math

def haversine_dist(lat1, lon1, lat2, lon2):
    """
    Calculates the distance (in km) between two points given their coordinates (latitude and longitude).
    """
    # Earth radius (in km)
    R = 6371.0

    lat1_rad = math.radians(lat1)
    lon1_rad = math.radians(lon1)
    lat2_rad = math.radians(lat2)
    lon2_rad = math.radians(lon2)

    dlat = lat2_rad - lat1_rad
    dlon = lon2_rad - lon1_rad

    # Haversine formula
    a = (math.sin(dlat / 2)**2) + (math.cos(lat1_rad) * math.cos(lat2_rad) * math.sin(dlon / 2)**2)
    
    # c is the angular distance in radians
    c = 2 * math.atan2(math.sqrt(a), math.sqrt(1 - a))

    # Return distance
    return R * c

import json

def sevici_disponibles_cerca(file, latitude, longitude, distance_km):
    station_list = []

    with open(file, "r",encoding="utf-8") as json_file:
        data = json.load(json_file)
    data = data["network"]["stations"]

    for station in data:
        station_info = []
        dist = haversine_dist(latitude, longitude, station["latitude"], station["longitude"])

        if dist <= distance_km:
            station_info.append(dist)
            station_info.append(station["name"])
            station_info.append((station["latitude"], station["longitude"]))
            station_info.append(station["free_bikes"])
            station_list.append(station_info)

    return sorted(station_list, key=lambda row: row[0])

 <p>La imputacin de series temporales multivariantes es una tarea fundamentalen el mbito del aprendizaje automtico, especialmente en aplicaciones &gt;sensibles como la medicina o la monitorizacin industrial. La presencia de valores ausentes en datos secuenciales puede comprometer gravemente la calidad de los modelos predictivos, dificultando la toma de decisiones informadas y, en contextos crticos, poniendo en riesgo la vida de las personas.</p>
 <p>Esta tesis doctoral aborda este desafo desde una perspectiva generativa, explorando el potencial de distintas arquitecturas para resolver el problema de la imputacin con precisin, eficiencia y escalabilidad.</p>
 <p>El trabajo comienza con un estudio centrado en la prediccin temprana de sepsis en unidades de cuidados intensivos (UCI), utilizando el dataset MIMIC-III. En este anlisis se demuestra cmo la eleccin de la estrategia de imputacin influye de forma significativa en el rendimiento de los modelos predictivos, especialmente en tareas clnicas donde la calidad y la fiabilidad de los datos pueden marcar la diferencia entre una intervencin eficaz o tarda. Este primer estudio sirve como justificacin y punto de partida para el desarrollo de nuevas propuestas que aborden el problema de la imputacin desde un enfoque generativo.</p>
 <p>A lo largo de la tesis se presentan tres propuestas metodolgicas que reflejan una evolucin progresiva en cuanto a sofisticacin tcnica y adecuacin a escenarios reales: G-TIGRE, un modelo basado en redes generativas adversariales y redes neuronales de grafos; TIMBA, que introduce modelos de difusin y bloques Mamba bidireccionales; y finalmente CoSTI, la primera adaptacin conocida de los Consistency Models al mbito de la imputacin espacio-temporal. Este ltimo modelo, entrenado mediante la tcnica de Consistency Training, es capaz de realizar imputaciones precisas en un nico paso de inferencia, lo que representa una reduccin del coste computacional de hasta un 98% frente a modelos de difusin tradicionales.</p>

 <p>Esta tesis se aborda desde el campo de la ingeniera neuromrfica, cuyo objetivoes emular la eficiencia computacional y energtica de los sistemas neuronalesbiolgicos para superar las limitaciones de las arquitecturas de computacin digitales.  <p>El trabajo se centra en el diseo y la implementacin de un marco para el procesamiento de informacin pulsante en diferentes partes del flujo de procesado de esta, desde la captura de datos hasta la actuacin. Se explora el uso de sensores neuromrficos avanzados como las retinas de silicio (DVS, DAVIS) y ccleas de silicio (NAS), que replican el comportamiento de los rganos sensoriales biolgicosy emplean el protocolo Address Event Representation (AER).  <p>Las contribuciones principales de esta tesis incluyen la grabacin y publicacin de un dataset multimodal de lectura de labios pulsante, LIPSFUS, que incluye la transformacin a pulsante de un dataset de lectura de labios denominado LRW a su versin pulsante (LRW-FUS) as como de un dataset multimodal de trayectorias robticas que incluye informacin posicional de un brazo robtico junto con informacin pulsante proveniente de su controladora, que implementa un control Spiking Proportional-Integrative-Derivative (SPID). Tambin se presenta un modelo basado en convoluciones en 1D con el que se ha logrado obtener hasta un 88% de precisin en la clasificacin de los datos de un subconjunto del dataset LIPSFUS, denominado LRW-FUS. Para finalizar, tambin se han desarrollado avances para la utilizacin de un brazo robtico con una controladora pulsante, la plataforma EDScorbot, con la cual se ha grabado el dataset de trayectorias robticasanteriormente mencionado.</p>
 <p>Los resultados demuestran la viabilidad de la integracin de arquitecturas de computacin pulsantes tanto en sistemas robticos complejos como entornos de captura de informacin y el procesado de esta, sentando las bases para unanueva generacin de sistemas inteligentes y eficientes. La funcin debe hacer scraping de la web de Marca (http://www.marca.com/futbol/primera/calendario.html), y almacenar en el fichero en formato CSV la informacin de aquellos partidos en los que el equipo indicado haya marcado $n$ o ms goles. import csv

def mas_de_n_goles(equipo, n, fichero):
    url = "https://www.marca.com/futbol/primera-division/calendario.html"
    soup = BeautifulSoup(requests.get(url).text)
    jornadas = soup.find_all("div", "jornada")

    jornada_dict ={}

    # Creamos un diccionario con las jornadas y sus partidos
    for jornada in jornadas:
        nombre_jornada = jornada.find("caption").text
        partidos_jornada = jornada.find_all("tr")
        jornada_dict[nombre_jornada] = partidos_jornada

    # Para cada partido en una jornada, obtenemos quien es el local, visitante, y resultado
    
    for nombre_jornada, partidos_jornada in jornada_dict.items():
        results_list = []
        for partido_jornada in partidos_jornada[1:]:
            local = visitante = resultado = ""
            try:
                local = partido_jornada.find("td", "local").text.strip()
                visitante = partido_jornada.find("td", "visitante").text.strip()
                resultado = partido_jornada.find("td", "resultado").text.strip()
                try:
                    goles_local, goles_visitante = resultado.split('-')
                    if (local == equipo and int(goles_local) >= n) or (visitante == equipo and int(goles_visitante) >= n):
                        results_list.append((local, visitante, resultado))
                except:
                    continue 
            except:
                pass
        # Eliminamos la primera posicin, que solo tena el header de la tabla
        jornada_dict[nombre_jornada] = results_list 

    dict_keys = list(jornada_dict.keys())
    for key in dict_keys:
        if len(jornada_dict[key]) == 0:
            del jornada_dict[key]
    
    with open(fichero, mode='w', newline='') as file:
        writer = csv.writer(file)
        writer.writerow(['Jornada', 'Local','Visitante','Resultado'])
        
        for jornada, info in jornada_dict.items():
            jornada_name = jornada
            for i in range(len(info)):
                local, visitante, resultado = info[i]
                writer.writerow([jornada_name, local, visitante, resultado])
            
    return jornada_dict

def harry_potter():
    """
        Obtiene las pelculas de Harry Potter y sus puntuaciones IMDB,
        devolviendo un DataFrame ordenado por puntuacin descendente.
    
        Returns:
            pd.DataFrame: DataFrame con columnas 'Movie' y 'Score', ordenado por 'Score' descendente.

    url = f"http://www.omdbapi.com/"
    params = {
        'apikey': OMDb_API_KEY,
        's': 'Harry Potter'
    }
    response = requests.get(url, params=params)
    data = response.json()
    movies = []

            'Score': float(movie_data['imdbRating'])
        })
    df_movies = pd.DataFrame(movies)
    df_movies.sort_values(by='Score', ascending=False, inplace=True)

import requests

def info_pokemon(nombre: str):
    """
    Obtiene informacin de un Pokmon dado su nombre.
    Args:
        nombre (str): Nombre del Pokmon.
    Returns:
        tuple: (nombre_pokemon, tipos, peso, altura, n_moves) o None si no existe.
    """
    url = f"https://pokeapi.co/api/v2/pokemon/{nombre.lower()}"
    response = requests.get(url)

    if response.status_code == 200:
        data = response.json()
        nombre_pokemon = data['name'].capitalize()
        tipos = [tipo['type']['name'] for tipo in data['types']]
        peso = data['weight'] / 10.0  # Convertir a kg
        altura = data['height'] / 10.0  # Convertir a m
        moves = data['moves']
        n_moves = len(moves)
        
        return (nombre_pokemon, tipos, peso, altura, n_moves)
    else:
        return None

def dataset_pokemon(lista_pokemons: list):
    """
